### 迭代2: 动态仓位引擎（修复A1）

```python
def dynamic_position_adjustment(confidence: float, regime: str) -> float:
    """根据置信度和市场状态动态调整仓位"""
    base_position = 20.0  # 基础仓位20%
    
    # 置信度调整
    if confidence >= 90:
        conf_adj = 1.5  # 满配
    elif confidence >= 80:
        conf_adj = 1.25
    elif confidence >= 70:
        conf_adj = 1.0
    else:
        conf_adj = 0.75
    
    # 市场状态调整
    if regime == "bull":
        regime_adj = 1.25
    elif regime == "neutral":
        regime_adj = 1.0
    else:  # bear/volatile
        regime_adj = 0.75
    
    position = base_position * conf_adj * regime_adj
    return min(position, 30.0)  # 上限30%
```
