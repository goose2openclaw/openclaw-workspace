```yaml
# aliyun-security-config.yaml
security:
  # 1. 安全组规则
  security_group:
    inbound:
      - port: 443
        source: 0.0.0.0/0
        protocol: HTTPS
      - port: 22
        source: YOUR_IP/32
        protocol: SSH
    outbound:
      - port: 443
        destination: 0.0.0.0/0
        protocol: HTTPS
      
  # 2. RAM访问控制
  ram:
    users:
      - name: deploy-bot
        policies:
          - AliyunECSReadOnlyAccess
          - AliyunVPCReadOnlyAccess
    mfa_enabled: true
    
  # 3. 密钥管理
  kms:
    enabled: true

  # 4. 日志
  logs:
    action_trail: true

  # 5. 告警

    - type: cpu
      threshold: 80%
      action: notify
    - type: memory
      threshold: 85%
```

### 6.2 一键部署脚本

```bash
#!/bin/bash
# 阿里云一键部署脚本 - 标准版

# 配置
REGION="ap-southeast-1"  # 新加坡
INSTANCE_TYPE="ecs.t6-c1m2.large"  # 2核4G
IMAGE="ubuntu_22_04_64_20G_alibaba_20231122.vhd"
PASSWORD="YourSecurePassword123!"

# 1. 创建VPC
aliyun vpc CreateVpc \
  --RegionId $REGION \
  --VpcName "go2se-vpc" \
  --CidrBlock "10.0.0.0/16"

# 2. 创建安全组
aliyun ecs CreateSecurityGroup \
  --RegionId $REGION \
  --SecurityGroupName "go2se-sg" \
  --VpcId "vpc-xxx"

# 3. 安全组规则
aliyun ecs AuthorizeSecurityGroup \
  --RegionId $REGION \
  --SecurityGroupId "sg-xxx" \
  --IpProtocol "tcp" \
  --PortRange "443/443" \
  --SourceCidrIp "0.0.0.0/0"

# 4. 创建ECS实例
aliyun ecs CreateInstance \
  --RegionId $REGION \
  --ImageId $IMAGE \
  --InstanceType $INSTANCE_TYPE \
  --SecurityGroupId "sg-xxx" \
  --VSwitchId "vsw-xxx" \
  --InstancePassword $PASSWORD

# 5. 安装Docker
ssh root@instance_ip "apt update && apt install -y docker.io docker-compose"

# 6. 部署应用
scp docker-compose.yml root@instance_ip:/root/
ssh root@instance_ip "cd /root && docker-compose up -d"

echo "部署完成! 访问 https://your-domain.com"
```
